34inline msgs::SphericalCoordinatesType ConvertCoord(
35 const math::SphericalCoordinates::CoordinateType &_sc)
36{
37 auto result = msgs::SphericalCoordinatesType::LOCAL2;
38 switch (_sc)
39 {
40 case math::SphericalCoordinates::CoordinateType::ECEF:
41 result = msgs::SphericalCoordinatesType::ECEF;
42 break;
43 case math::SphericalCoordinates::CoordinateType::GLOBAL:
44 result = msgs::SphericalCoordinatesType::GLOBAL;
45 break;
46 case math::SphericalCoordinates::CoordinateType::SPHERICAL:
47 result = msgs::SphericalCoordinatesType::SPHERICAL;
48 break;
49 case math::SphericalCoordinates::CoordinateType::LOCAL:
50 result = msgs::SphericalCoordinatesType::LOCAL;
51 break;
52 case math::SphericalCoordinates::CoordinateType::LOCAL2:
53 result = msgs::SphericalCoordinatesType::LOCAL2;
54 break;
55 default:
56 std::cerr << "Invalid coordinate type passed" << std::endl;
57 }
58 return result;
59}

62inline math::SphericalCoordinates::CoordinateType Convert(
63 const msgs::SphericalCoordinatesType &_sc)
64{
65 switch (_sc)
66 {
67 case msgs::SphericalCoordinatesType::ECEF:
68 return math::SphericalCoordinates::CoordinateType::ECEF;
69 case msgs::SphericalCoordinatesType::GLOBAL:
70 return math::SphericalCoordinates::CoordinateType::GLOBAL;
71 case msgs::SphericalCoordinatesType::SPHERICAL:
72 return math::SphericalCoordinates::CoordinateType::SPHERICAL;
73 case msgs::SphericalCoordinatesType::LOCAL:
74 return math::SphericalCoordinates::CoordinateType::LOCAL;
75 case msgs::SphericalCoordinatesType::LOCAL2:
76 return math::SphericalCoordinates::CoordinateType::LOCAL2;
77 default:
78 std::cerr << "Invalid coordinate type passed" << std::endl;
79 }
80 return math::SphericalCoordinates::CoordinateType::LOCAL2;
81}

84inline void Set(gz::msgs::SphericalCoordinates *_msg,
85 const gz::math::SphericalCoordinates &_data)
86{
87 if (_data.Surface() == math::SphericalCoordinates::EARTH_WGS84)
88 {
89 _msg->set_surface_model(msgs::SphericalCoordinates::EARTH_WGS84);
90 }
91 else if (_data.Surface() == math::SphericalCoordinates::MOON_SCS)
92 {
93 _msg->set_surface_model(msgs::SphericalCoordinates::MOON_SCS);
94 }
95 else if (_data.Surface() ==
96 math::SphericalCoordinates::CUSTOM_SURFACE)
97 {
98 _msg->set_surface_model(
99 msgs::SphericalCoordinates::CUSTOM_SURFACE);
100 _msg->set_surface_axis_equatorial(_data.SurfaceAxisEquatorial());
101 _msg->set_surface_axis_polar(_data.SurfaceAxisPolar());
102 }
103 else
104 {
105 std::cerr << "Unrecognized spherical surface type ["
106 << _data.Surface()
107 << "]. Not populating message field." << std::endl;
108 }
109 _msg->set_latitude_deg(_data.LatitudeReference().Degree());
110 _msg->set_longitude_deg(_data.LongitudeReference().Degree());
111 _msg->set_elevation(_data.ElevationReference());
112 _msg->set_heading_deg(_data.HeadingOffset().Degree());
113}

115inline void Set(gz::math::SphericalCoordinates *_data,
116 const gz::msgs::SphericalCoordinates &_msg)
117{
118 if (_msg.surface_model() == msgs::SphericalCoordinates::EARTH_WGS84)
119 {
120 _data->SetSurface(math::SphericalCoordinates::EARTH_WGS84);
121 }
122 else if (_msg.surface_model() == msgs::SphericalCoordinates::MOON_SCS)
123 {
124 _data->SetSurface(math::SphericalCoordinates::MOON_SCS);
125 }
126 else if (_msg.surface_model() == msgs::SphericalCoordinates::CUSTOM_SURFACE)
127 {
128 _data->SetSurface(math::SphericalCoordinates::CUSTOM_SURFACE,
129 _msg.surface_axis_equatorial(),
130 _msg.surface_axis_polar());
131 }
132 else
133 {
134 std::cerr << "Unrecognized spherical surface type ["
135 << _msg.surface_model()
136 << "]. Not populating data field." << std::endl;
137 }
138
139 auto lat = gz::math::Angle();
140 lat.SetDegree(_msg.latitude_deg());
141 _data->SetLatitudeReference(lat);
142
143 auto lon = gz::math::Angle();
144 lon.SetDegree(_msg.longitude_deg());
145 _data->SetLongitudeReference(lon);
146
147 auto head = gz::math::Angle();
148 head.SetDegree(_msg.heading_deg());
149 _data->SetHeadingOffset(head);
150
151 _data->SetElevationReference(_msg.elevation());
152}
