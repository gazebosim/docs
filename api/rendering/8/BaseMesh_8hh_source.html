294  template <class T>
295  void BaseMesh<T>::SetMaterial(MaterialPtr _material, bool _unique)
296  {
297  // todo(anyone) take ownership of reference _material if _unique
298  // and destroy the reference material when the mesh is destroyed
299  unsigned int count = this->SubMeshCount();
300  _material = (_unique && count > 0) ? _material->Clone() : _material;
301 
302  for (unsigned int i = 0; i < count; ++i)
303  {
304  SubMeshPtr subMesh = this->SubMeshByIndex(i);
305  subMesh->SetMaterial(_material, false);
306  }
307 
308  // If the same material is being set, return early and don't try
309  // to destroy the material. We still need call SetMaterial on the
310  // submeshes in case the user changed some material properties
311  // before setting it back.
312  if (!_unique && _material == this->material)
313  return;
314 
315  if (this->material && this->ownsMaterial)
316  this->Scene()->DestroyMaterial(this->material);
317 
318  this->ownsMaterial = _unique;
319  this->material = _material;
320  }

338  template <class T>
339  GeometryPtr BaseMesh<T>::Clone() const
340  {
341  if (!this->Scene())
342  {
343  gzerr << "Cloning a mesh failed because the mesh to be "
344  << "cloned does not belong to a scene.\n";
345  return nullptr;
346  }
347  else if (this->meshDescriptor.meshName.empty())
348  {
349  gzerr << "Cloning a geometry failed because the name of the mesh is "
350  << "missing.\n";
351  return nullptr;
352  }
353 
354  auto result = this->Scene()->CreateMesh(this->meshDescriptor);
355  if (result)
356  {
357  if (this->Material())
358  {
359  // this call will set the material for the mesh and its submeshes
360  result->SetMaterial(this->Material());
361  }
362  else
363  {
364  // if the mesh doesn't have a material, clone any existing submesh
365  // materials
366  for (unsigned int i = 0; i < this->SubMeshCount(); ++i)
367  {
368  auto existingSubMeshMaterial = this->SubMeshByIndex(i)->Material();
369  if (existingSubMeshMaterial)
370  result->SubMeshByIndex(i)->SetMaterial(existingSubMeshMaterial);
371  }
372  }
373  }
374 
375  return result;
376  }
